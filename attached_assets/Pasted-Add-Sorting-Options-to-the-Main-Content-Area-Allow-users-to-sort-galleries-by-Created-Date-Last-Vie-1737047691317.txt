Add Sorting Options to the Main Content Area
Allow users to sort galleries by Created Date, Last Viewed, or Alphabetical Order.

Instructions:
Update MainContentV2 Component:

Add a dropdown for sorting above the grid of gallery cards.
Store the selected sort order in a useState variable.
Sort the galleries client-side based on the selected option.

import React, { useState } from "react";
import { useQuery } from "@tanstack/react-query";

export const MainContentV2 = () => {
  const [sortOrder, setSortOrder] = useState("created"); // Default sort: Created Date
  const { data: galleries = [], isLoading } = useQuery({
    queryKey: ["/api/galleries"],
    queryFn: async () => {
      const res = await fetch("/api/galleries");
      if (!res.ok) throw new Error("Failed to fetch galleries");
      return res.json();
    },
  });

  const sortedGalleries = [...galleries].sort((a, b) => {
    if (sortOrder === "created") return new Date(b.createdAt) - new Date(a.createdAt);
    if (sortOrder === "lastViewed") return new Date(b.lastViewedAt) - new Date(a.lastViewedAt);
    if (sortOrder === "alphabetical") return a.title.localeCompare(b.title);
    return 0;
  });

  if (isLoading) {
    return <div>Loading...</div>;
  }

  return (
    <div>
      {/* Sorting Dropdown */}
      <div className="flex justify-end mb-4">
        <select
          className="border border-gray-300 rounded px-4 py-2"
          value={sortOrder}
          onChange={(e) => setSortOrder(e.target.value)}
        >
          <option value="created">Created Date</option>
          <option value="lastViewed">Last Viewed</option>
          <option value="alphabetical">Alphabetical</option>
        </select>
      </div>

      {/* Gallery Grid */}
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-6">
        {sortedGalleries.map((gallery) => (
          <div
            key={gallery.id}
            className="group bg-white shadow-md rounded-lg overflow-hidden"
          >
            <div className="aspect-w-4 aspect-h-3 bg-gray-200">
              <img
                src={gallery.thumbnailUrl || ""}
                alt={gallery.title}
                className="object-cover w-full h-full group-hover:scale-105 transition-transform"
              />
            </div>
            <div className="p-4">
              <h3 className="font-semibold text-lg">{gallery.title}</h3>
              <p className="text-sm text-gray-600">{gallery.imageCount} images</p>
            </div>
          </div>
        ))}
      </div>
    </div>
  );
};
