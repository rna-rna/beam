Goal: Clicking a folder should display the contents of that folder (even if it's currently empty).

Instructions:
Add State for Current Folder:

Track which folder is currently open using useState.
Update Folder Click Handler:

When a folder is clicked, set it as the currentFolder.
Modify Gallery Rendering Logic:

Only display galleries that belong to the currentFolder.

import React, { useState } from "react";

export const MainContentV2 = () => {
  const [currentFolder, setCurrentFolder] = useState<string | null>(null); // Track the current folder
  const { data: galleries = [] } = useQuery({
    queryKey: ["/api/galleries"],
    queryFn: async () => {
      const res = await fetch("/api/galleries");
      if (!res.ok) throw new Error("Failed to fetch galleries");
      return res.json();
    },
  });

  const { data: folders = [] } = useQuery({
    queryKey: ["/api/folders"],
    queryFn: async () => {
      const res = await fetch("/api/folders");
      if (!res.ok) throw new Error("Failed to fetch folders");
      return res.json();
    },
  });

  const displayedGalleries = currentFolder
    ? galleries.filter((gallery) => gallery.folderId === currentFolder)
    : galleries; // Show all galleries if no folder is selected

  return (
    <div className="flex h-full">
      {/* Sidebar */}
      <div className="w-64 bg-white border-r">
        <div className="p-4">
          <button
            onClick={() => setCurrentFolder(null)} // Reset to show all galleries
            className="w-full text-left text-blue-600"
          >
            Recents
          </button>
        </div>
        <ul className="p-4 space-y-2">
          {folders.map((folder) => (
            <li
              key={folder.id}
              className="cursor-pointer px-4 py-2 bg-gray-100 rounded"
              onClick={() => setCurrentFolder(folder.id)} // Open the selected folder
            >
              {folder.name}
            </li>
          ))}
        </ul>
      </div>

      {/* Main Content */}
      <div className="flex-1 px-6 py-8">
        <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-6">
          {displayedGalleries.length > 0 ? (
            displayedGalleries.map((gallery) => (
              <div key={gallery.id} className="bg-white shadow-md rounded-lg">
                {/* Render gallery card */}
                <h3>{gallery.title}</h3>
              </div>
            ))
          ) : (
            <p>No galleries in this folder.</p>
          )}
        </div>
      </div>
    </div>
  );
};
