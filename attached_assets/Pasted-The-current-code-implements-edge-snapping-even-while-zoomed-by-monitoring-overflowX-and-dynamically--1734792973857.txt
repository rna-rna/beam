const handlePan = (event: any, info: PanInfo) => {
  const scale = scaleValue.get();
  const maxX = (window.innerWidth / 2) * (scale - 1);
  const maxY = (window.innerHeight / 2) * (scale - 1);

  const newX = offsetX.get() + info.delta.x;
  const newY = offsetY.get() + info.delta.y;

  // If zoomed in, disable snapping to next image
  if (scale > 1) {
    offsetX.set(clampPan(newX, maxX));
    offsetY.set(clampPan(newY, maxY));
  } else {
    // Allow swipe to next image if at default zoom
    const overflowX = Math.abs(newX) - maxX;
    const dynamicThreshold = scale > 1 ? 9999 : 150;  // Prevent snapping while zoomed

    if (overflowX > dynamicThreshold) {
      offsetX.set(newX * 1.2);
      setTimeout(() => {

        const clampedIndex = Math.max(0, Math.min(nextIndex, images.length - 1));

How This Works:
Zoom Isolation: If scale > 1, the image cannot trigger snapping to the next image, regardless of horizontal pan (offsetX).
Pan Clamping: The offsetX and offsetY are clamped within the image bounds (maxX and maxY). This prevents dragging beyond the image's edges.
Transition Unlocking: When zoomed out (scale === 1), normal swiping to the next image resumes.
