3. Add the Settings Link to the Avatar Dropdown:
Open your Navbar.tsx or layout component (where the user avatar is).

Add a dropdown link that directs users to /settings:

tsx
Copy code
import { UserButton } from '@clerk/clerk-react';
import { useNavigate } from 'react-router-dom';
import { useState } from 'react';

export default function Navbar() {
  const navigate = useNavigate();
  const [dropdownOpen, setDropdownOpen] = useState(false);

  return (
    <div className="flex items-center gap-4 relative">
      <UserButton
        afterSignOutUrl="/"
        appearance={{ elements: { avatarBox: 'h-10 w-10' } }}
        onClick={() => setDropdownOpen(!dropdownOpen)}
      />
      
      {dropdownOpen && (
        <div className="absolute top-12 right-0 w-48 bg-white rounded-lg shadow-lg">
          <button
            onClick={() => navigate('/settings')}
            className="block px-4 py-2 text-sm text-gray-700 hover:bg-gray-100 w-full text-left"
          >
            Profile Settings
          </button>
        </div>
      )}
    </div>
  );
}
